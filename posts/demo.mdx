  const [copy, isCopied] = useCopyToClipboard();

  return (
    <Button
      onClick={() =>
        copy("hello world").then(() => console.info("Text copied!"))
      }
    >
      {isCopied ? "Text Copied" : "Click to copy"}
    </Button>
  );
}
